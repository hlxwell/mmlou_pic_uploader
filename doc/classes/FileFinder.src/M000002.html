  def all_directories
    full_directories = []
    directories = Dir.entries(@root)
    
    #remove . and .. dir
    [".",".."].each do |dir|
      directories.delete(dir)
    end

    directories.each do |dir|
      if File.directory?(@root+dir)
        full_directories << @root+dir
      end
    end
    return full_directories
  end
